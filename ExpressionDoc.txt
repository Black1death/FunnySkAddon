Expressions:
	guild in position %integer%:
		Opis: Pokazuje pozycje gildi w topce (topka aktualizowana jest co 5 minut lub poleceniem "update guild tops")
		Przykłady użycia:
			send "Gildia na pozycji 1: %guild in position 1%"
	player in position %integer%:
		Opis: Pokazuje pozycje gracza w topce (topka aktualizowana jest co 5 minut lub poleceniem "update guild tops")
		Przykłady użycia:
			send "Gracz na pozycji 1: %player in position 1%"
	guild name from tag %string%:
		Opis: Pokazuje nazwe gildii z podanego tagu
		Przykłady użycia:
			send "Nazwa gildi o tagu TAG to %guild name from tag "TAG"%"
	guild from tag %string%:
		Opis: Zwraca gildie po podanym tagu
		Przykłady użycia:
			set {_guild} to guild from tag "TAG"
	guild from name %string%:
		Opis: Zwraca gildie po podanej nazwie
		Przykłady użycia:
			set {_guild} to guild from name "NAZWAGILDI"	
	guild from player %player%:
		Opis: Zwraca gildie po podanym graczu
		Przykłady użycia:
			set {_guild} to guild from player player
	%guild/text/player% guild deaths:
		Opis: Zwraca śmierci gildi
		Przykłady użycia:
			set {_guildDeaths} to "NAZWAGILDI" guild deaths
		Uwagi: W miejsce "NAZWAGILDI" można dać obiekt gildii zapisany np. w zmiennej  lub gracza którego ma być to gildia
